  def [](key)
    val = @contents_hash[@first_level][key]
    if val
      return val
    else
      default_val = @contents_hash["default"][val]
      if default_val
        return default_val
      else
        debug_p "key( #{key.to_s} ) is not found on even default hash!"
      end
    end
  end
